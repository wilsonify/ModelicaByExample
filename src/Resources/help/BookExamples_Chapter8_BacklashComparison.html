Modelica definition

partial model NoRigid
  import Modelica.Mechanics.Rotational;
  
  Rotational.Inertia I1;
  Rotational.Inertia I2(J=3);
  Rotational.Torque torque;
  Modelica.Blocks.Sources.Sine signal;
  replaceable Rotational.Interfaces.Compliant backlash;
equation 
  connect(signal.outPort, torque.inPort);
  connect(torque.flange_b, I1.flange_a);
  connect(I1.flange_b, backlash.flange_a);
  connect(backlash.flange_b, I2.flange_a);
end NoRigid;
