model ChemicalSystem
  extends BookExamples.Icons.RunnableExample;
  Chemistry.Basic.Volume v(
    nspecies=nspecies,
    v=1,
    i_moles=fill(1, nspecies));
  Reactions.R_AY r_ay(nspecies=nspecies, v=1);
  Reactions.R_XY r_xy(nspecies=nspecies, v=1);
  Reactions.R_AX r_ax(nspecies=nspecies, v=1);
  Reactions.R_2X r_2x(nspecies=nspecies, v=1);
  Reactions.R_BZ r_bz(nspecies=nspecies, v=1);
  Chemistry.Basic.Stationary c_A(
    stat_species=A,
    nspecies=nspecies,
    c=1.0);
  Chemistry.Basic.Stationary c_B(
    stat_species=B,
    nspecies=nspecies,
    c=1.0);
  
  Real cA=v.moles[A];
  Real cB=v.moles[B];
  Real cP=v.moles[P];
  Real cX=v.moles[X];
  Real cY=v.moles[Y];
  Real cZ=v.moles[Z];
protected 
  Real dcx;
  Real dcy;
  Real dcz;
equation 
  connect(v.p, r_ay.p);
  connect(v.p, r_xy.p);
  connect(v.p, r_ax.p);
  connect(v.p, r_2x.p);
  connect(v.p, r_bz.p);
  connect(v.p, c_A.p);
  connect(v.p, c_B.p);
  dcx = der(cX);
  dcy = der(cY);
  dcz = der(cZ);
end ChemicalSystem;

model ODESystem
  extends BookExamples.Icons.RunnableExample;
  
  Real x(start=1);
  Real y(start=1);
  Real z(start=1);
  Real dx;
  Real dy;
  Real dz;
  constant Real alpha=50;
  constant Real beta=1.0;
  constant Real gamma=0.0013;
  constant Real f=2;
  Real rate_xy;
  Real rate_2x;
  Real rate_ax;
  Real rate_ay;
  Real rate_bz;
equation 
  rate_ay = y;
  rate_xy = x*y;
  rate_ax = alpha*x;
  rate_2x = 2*gamma*x*x;
  rate_bz = beta*z;
  dx = -x*y - 2*gamma*x*x + alpha*x + y;
  dy = -x*y - y + (f/2)*beta*z;
  dz = 2*alpha*x - beta*z;
  der(x) = -x*y - 2*gamma*x*x + alpha*x + y;
  der(y) = -x*y - y + (f/2)*beta*z;
  der(z) = 2*alpha*x - beta*z;
end ODESystem;
